301 template <>
302 struct VisitPreorderImpl<PPUMemoryBlock>
303 {
304  template <typename ContainerT, typename VisitorT>
305  static void call(
306  ContainerT& config, PPUMemoryBlock::coordinate_type const& coord, VisitorT&& visitor)
307  {
308  using namespace halco::hicann_dls::vx;
309  auto const ppu_coord = coord.toPPUOnDLS();
310 
311  if (coord.toPPUMemoryBlockOnPPU().toPPUMemoryBlockSize() != config.size()) {
312  std::stringstream ss;
313  ss << "container size(" << config.size() << ") and " << coord << " size do not match.";
314  throw std::runtime_error(ss.str());
315  }
316 
317  visitor(coord, config);
318 
319  for (size_t counter = 0; counter < config.size(); counter++) {
320  auto word_coord = PPUMemoryWordOnDLS(
321  PPUMemoryWordOnPPU(coord.toPPUMemoryBlockOnPPU().toMin() + counter), ppu_coord);
322  visit_preorder(config.m_words.at(counter), word_coord, visitor);
323  }
324  }
325 };
